# 高度可自定义博客模版 — 实现计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**目标：** 构建一个高度可自定义的博客模版，具有管理员后台，支持对博客的所有元素进行自定义配置。

**架构概览：** 基于 Next.js 15 (App Router) 全栈架构，前台展示端使用 SSR/SSG 渲染，管理后台通过 API 路由与 Prisma ORM 交互，数据存储在 SQLite 中，支持未来迁移到 PostgreSQL。前台采用瀑布流布局 + 极简杂志混搭风格。

**技术栈：** Next.js 15 (App Router) / Prisma + SQLite / Tailwind CSS / @uiw/react-md-editor + TipTap / Vercel

```typescript
// src/lib/auth.ts
import bcrypt from 'bcryptjs';
import jwt from 'jsonwebtoken';
import { cookies } from 'next/headers';
import { prisma } from './prisma';

const AUTH_SECRET = process.env.AUTH_SECRET || 'secret';
const COOKIE_NAME = 'admin_token';

export async function verifyPassword(input: string, hashed: string): Promise<boolean> {
  return bcrypt.compare(input, hashed);
}

export async function hashPassword(password: string): Promise<string> {
  return bcrypt.hash(password, 10);
}

export function generateToken(id: number): string {
  return jwt.sign({ id }, AUTH_SECRET, { expiresIn: '7d' });
}

export function verifyToken(token: string): { id: number } | null {
  try {
    return jwt.verify(token, AUTH_SECRET) as { id: number };
  } catch {
    return null;
  }
}

export async function getAdminFromCookies(): Promise<{ id: number } | null> {
  const cookieStore = await cookies();
  const token = cookieStore.get(COOKIE_NAME)?.value;
  if (!token) return null;
  return verifyToken(token);
}

export async function loginAdmin(password: string): Promise<{ success: boolean; token?: string; error?: string }> {
  const admin = await prisma.admin.findFirst();
  if (!admin) return { success: false, error: '管理员未初始化' };
  const valid = await verifyPassword(password, admin.password);
  if (!valid) return { success: false, error: '密码错误' };
  const token = generateToken(admin.id);
  return { success: true, token };
}
```

```typescript
// src/lib/settings.ts
'use server';
import { prisma } from './prisma';
import { cache } from 'react';

export const getSiteSettings = cache(async () => {
  const settings = await prisma.siteSettings.findUnique({ where: { id: 1 } });
  if (!settings) {
    return {
      themeColors: { primary: '#3B82F6', background: '#FFFFFF', text: '#1F2937' },
      fonts: { body: 'Noto Sans SC', heading: 'Noto Serif SC' },
      layoutConfig: { sidebarPosition: 'right', navStyle: 'sticky' },
      homepageModules: [],
      seoConfig: { title: '我的博客', description: '', keywords: '' },
      customCSS: '',
    };
  }
  return {
    themeColors: JSON.parse(settings.themeColors),
    fonts: JSON.parse(settings.fonts),
    layoutConfig: JSON.parse(settings.layoutConfig),
    homepageModules: JSON.parse(settings.homepageModules),
    seoConfig: JSON.parse(settings.seoConfig),
    customCSS: settings.customCSS,
  };
});

export async function updateSiteSettings(data: Partial<{
  themeColors: object;
  fonts: object;
  layoutConfig: object;
  homepageModules: object[];
  seoConfig: object;
  customCSS: string;
}>) {
  const updates: Record<string, string> = {};
  if (data.themeColors) updates.themeColors = JSON.stringify(data.themeColors);
  if (data.fonts) updates.fonts = JSON.stringify(data.fonts);
  if (data.layoutConfig) updates.layoutConfig = JSON.stringify(data.layoutConfig);
  if (data.homepageModules) updates.homepageModules = JSON.stringify(data.homepageModules);
  if (data.seoConfig) updates.seoConfig = JSON.stringify(data.seoConfig);
  if (data.customCSS !== undefined) updates.customCSS = data.customCSS;
  await prisma.siteSettings.update({ where: { id: 1 }, data: updates });
}
```

```typescript
export function slugify(text: string): string {
  return text
    .toLowerCase()
    .replace(/[^\w\s-]/g, '')
    .replace(/\s+/g, '-')
    .replace(/--+/g, '-')
    .trim();
}
```

```typescript
import { useState } from 'react';

interface Comment {
  id: number;

  content: string;
  createdAt: string;
}

interface CommentSectionProps {
  articleId: number;
  comments: Comment[];
}

export function CommentSection({ articleId, comments }: CommentSectionProps) {
  const [name, setName] = useState('');
  const [email, setEmail] = useState('');
  const [content, setContent] = useState('');
  const [submitting, setSubmitting] = useState(false);
  const [msg, setMsg] = useState('');

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!name.trim() || !content.trim()) return;
    setSubmitting(true);
    try {
      const res = await fetch('/api/comments', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ articleId, authorName: name, email, content }),
      });
      const data = await res.json();
      setMsg(res.ok ? '评论已提交，等待审核！' : (data.error || '提交失败'));
      if (res.ok) { setName(''); setEmail(''); setContent(''); }
    } catch { setMsg('网络错误'); }
    setSubmitting(false);
  };

  const approved = comments;

  return (
```

```typescript
// src/app/api/auth/login/route.ts
import { NextResponse } from 'next/server';
import { prisma } from '@/lib/prisma';
import bcrypt from 'bcryptjs';

export async function POST(request: Request) {
  try {
    const { password } = await request.json();
    const admin = await prisma.admin.findFirst();
    if (!admin) return NextResponse.json({ error: '管理员未初始化' }, { status: 401 });
    const valid = await bcrypt.compare(password, admin.password);
    if (!valid) return NextResponse.json({ error: '密码错误' }, { status: 401 });
    const token = Buffer.from(`${admin.id}:${admin.password}`).toString('base64');
    const response = NextResponse.json({ success: true });
    response.cookies.set('admin_token', token, {
      httpOnly: true,
      secure: process.env.NODE_ENV === 'production',
      sameSite: 'lax',
      maxAge: 60 * 60 * 24 * 7,
      path: '/',
    });
    return response;
  } catch {
    return NextResponse.json({ error: '服务器错误' }, { status: 500 });
  }
}
```

```typescript
import { prisma } from '@/lib/prisma';
```

```typescript
export async function GET(request: Request) {
  const { searchParams } = new URL(request.url);
  const status = searchParams.get('status');
  const articles = await prisma.article.findMany({
    where: status ? { status } : undefined,
    include: { categories: true, tags: true },
    orderBy: { createdAt: 'desc' },
  });
  return NextResponse.json(articles);
}

export async function POST(request: Request) {
  try {
    const body = await request.json();
    const { title, content, excerpt, coverImage, status, categoryIds, tagIds } = body;
    if (!title || !content) return NextResponse.json({ error: '标题和内容不能为空' }, { status: 400 });
    const slug = slugify(title) + '-' + Date.now().toString(36);
    const article = await prisma.article.create({
      data: {
        title, slug, content, excerpt, coverImage,
        status: status || 'draft',
        authorId: 1,
        categories: categoryIds?.length ? { connect: categoryIds.map((id: number) => ({ id })) } : undefined,
        tags: tagIds?.length ? { connect: tagIds.map((id: number) => ({ id })) } : undefined,
      },
    });
    return NextResponse.json(article);
  } catch {
    return NextResponse.json({ error: '创建失败' }, { status: 500 });
  }
}
```

```typescript
import { NextResponse } from 'next/server';
import { prisma } from '@/lib/prisma';
import bcrypt from 'bcryptjs';

export async function PUT(request: Request) {
  try {
    const { oldPassword, newPassword } = await request.json();
    const admin = await prisma.admin.findFirst();
    if (!admin) return NextResponse.json({ error: '管理员未找到' }, { status: 404 });
    const valid = await bcrypt.compare(oldPassword, admin.password);
    if (!valid) return NextResponse.json({ error: '原密码错误' }, { status: 401 });
    const hashed = await bcrypt.hash(newPassword, 10);
    await prisma.admin.update({ where: { id: admin.id }, data: { password: hashed } });
    return NextResponse.json({ success: true });
  } catch { return NextResponse.json({ error: '修改失败' }, { status: 500 }); }
}
```

```typescript
// src/app/admin/password/page.tsx
'use client';
import { useState } from 'react';

export default function AdminPasswordPage() {
  const [old, setOld] = useState('');
  const [neu, setNeu] = useState('');
  const [confirm, setConfirm] = useState('');
  const [msg, setMsg] = useState('');
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (neu !== confirm) { setMsg('两次密码不一致'); return; }
    if (neu.length < 6) { setMsg('密码至少6位'); return; }
    setLoading(true);
    try {
      const res = await fetch('/api/auth/password', { method: 'PUT', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ oldPassword: old, newPassword: neu }) });
      const data = await res.json();
      setMsg(res.ok ? '密码修改成功！' : (data.error || '修改失败'));
      if (res.ok) { setOld(''); setNeu(''); setConfirm(''); }
    } catch { setMsg('网络错误'); }
    setLoading(false);
  };

  return (
    <div className="max-w-md space-y-4">
      <h1 className="text-2xl font-bold text-[var(--color-text)]">修改管理密码</h1>
      <form onSubmit={handleSubmit} className="bg-[var(--color-card)] p-5 rounded-xl border border-[var(--color-border)] space-y-4">
        {msg && <p className={`text-sm ${msg.includes('成功') ? 'text-green-500' : 'text-red-500'}`}>{msg}</p>}
        <input type="password" value={old} onChange={e => setOld(e.target.value)} placeholder="原密码" required
          className="w-full px-4 py-2 rounded-lg border border-[var(--color-border)] bg-[var(--color-background)] text-[var(--color-text)]" />
        <input type="password" value={neu} onChange={e => setNeu(e.target.value)} placeholder="新密码" required
          className="w-full px-4 py-2 rounded-lg border border-[var(--color-border)] bg-[var(--color-background)] text-[var(--color-text)]" />
        <input type="password" value={confirm} onChange={e => setConfirm(e.target.value)} placeholder="确认新密码" required
          className="w-full px-4 py-2 rounded-lg border border-[var(--color-border)] bg-[var(--color-background)] text-[var(--color-text)]" />
        <button type="submit" disabled={loading}
          className="w-full py-2 bg-[var(--color-primary)] text-white rounded-lg hover:opacity-90 transition-colors disabled:opacity-50">
          {loading ? '修改中...' : '修改密码'}
        </button>
      </form>
    </div>
  );
}
```
